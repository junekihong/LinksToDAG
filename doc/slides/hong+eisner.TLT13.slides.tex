\subsection{Connectedness, Acyclicity}
\frame{\frametitle{Connectedness, Acyclicity}
\begin{itemize}
  \item[] Connectedness
    \begin{align}
      \sum_u x_{uv} & \geq 1
    \end{align}

  \item[] Acyclicity
    \begin{itemize}
      \item a
    \end{itemize}

    \begin{align}
      (\forall_u)\; d_v + (1 + n_v) \cdot (1 - x_{uv}) & \geq 1+d_u
    \end{align}
\end{itemize}

}


\frame{\frametitle{Consistence of Directionalized Links}
\begin{itemize}
  \item[] Consistency of Directionalized Links
    \begin{align}\label{direction+slack}
      x_{ij} &\leq r_L &
      x_{ji} &\leq \ell_L 
    \end{align}

    \begin{align}\label{eqn:obj}
      \min \left( \sum_L r_L + \ell_L \right)
    \end{align}

    %$$x_{ij} \leq r_L + s_{ij} $$
    %$$x_{ji} \leq \ell_L + s_{ij}$$
    %$$\min \left( \sum_L r_L + \ell_L \right)$$

\end{itemize}
}

\frame{\frametitle{Consistence of Directionalized Links with Slack}
\begin{itemize}
  \item[] Consistency of Directionalized Links
    \begin{align}\label{direction+slack}
      x_{ij} &\leq r_L + s_{ij} &
      x_{ji} &\leq \ell_L + s_{ij} 
    \end{align}

    \begin{align}\label{eqn:obj}
      \min \left( \sum_L r_L + \ell_L \right) \cdot \frac{N_L}{4} + \sum_{ij}s_{ij}
    \end{align}
